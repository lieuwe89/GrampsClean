    ```python
    def get_event_place_handle(self, person, event_type):
        """
        Return the place handle (str) for a person's birth or death event,
        or None if no event, no place, or event_type not 'birth'/'death'.
        """
        try:
            if event_type == "birth":
                ref = person.get_birth_ref()
            elif event_type == "death":
                ref = person.get_death_ref()
            else:
                return None
            if ref is None:
                return None
            event = self.db.get_event_from_handle(ref.ref)
            if event is None:
                return None
            place_handle = event.get_place_handle()
            return place_handle if place_handle else None
        except Exception:
            return None
    ```

    pack result list; set _worker = None; set _snapshot = None.

    _build_toolbar(): create a Gtk.Box with:
      - Scan button (connect _on_scan)
      - Cancel button (initially insensitive, connect _on_cancel)
      - Separator or spacer
      - 4 Gtk.CheckButton widgets, all initially active=True:
        self._chk_birth_date  = Gtk.CheckButton(label="Birth Date")
        self._chk_death_date  = Gtk.CheckButton(label="Death Date")
        self._chk_birth_place = Gtk.CheckButton(label="Birth Place")
        self._chk_death_place = Gtk.CheckButton(label="Death Place")
      Pack buttons left-to-right: Scan, Cancel, then the 4 checkboxes.
      pack_start toolbar at top.

    _on_scan(btn):
      1. Guard: return if worker already running.
      2. Read checkbox states into a dict:
         selected = {
             "birth_date":  self._chk_birth_date.get_active(),
             "death_date":  self._chk_death_date.get_active(),
             "birth_place": self._chk_birth_place.get_active(),
             "death_place": self._chk_death_place.get_active(),
         }
         If no field is selected: set_status("Select at least one field."); return.
      3. result_list.clear(); set_status("Reading database…")
      4. Build snapshot on main thread:
         person_data = {}
         for person in db_wrap.iter_people():
             h = person.get_handle()
             person_data[h] = {
                 "name":        db_wrap.format_name(person),
                 "gramps_id":   db_wrap.get_gramps_id(person),
                 "birth_date":  db_wrap.get_event_date(person, "birth") is None,
                 "death_date":  db_wrap.get_event_date(person, "death") is None,
                 "birth_place": db_wrap.get_event_place_handle(person, "birth") is None,
                 "death_place": db_wrap.get_event_place_handle(person, "death") is None,
             }
         self._snapshot = {"person_data": person_data, "selected": selected}
      5. Disable Scan, enable Cancel, set_status("Scanning…")
      6. Start ScanWorker(result_list, _scan_fn, None); start worker.
      7. GLib.timeout_add(200, _check_done)

    _on_cancel(btn): worker.cancel() if worker; disable Cancel.

    _check_done(): return True (keep polling) if worker running;
      else enable Scan, disable Cancel; return False.

    _scan_fn(_db_wrap_unused): generator — NO DB access here.
      snap = self._snapshot
      person_data = snap["person_data"]
      selected = snap["selected"]

      FIELD_LABELS = {
          "birth_date":  "Birth Date",
          "death_date":  "Death Date",
          "birth_place": "Birth Place",
          "death_place": "Death Place",
      }

      rows = []
      for handle, info in person_data.items():
          missing = [
              FIELD_LABELS[f]
              for f in ("birth_date", "death_date", "birth_place", "death_place")
              if selected.get(f) and info[f]  # field selected AND missing
          ]
          if missing:
              rows.append((
                  info["name"],
                  info["gramps_id"],
                  ", ".join(missing),
                  handle,        # hidden tag for navigation (03-02)
              ))

      if not rows:
          yield ("", "", ""), "No missing data found for selected fields."
          return

      total = len(rows)
      for i, row in enumerate(rows):
          status = f"{total} people with missing data" if i == len(rows) - 1 else ""
          yield row, status
